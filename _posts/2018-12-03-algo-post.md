# [BOJ 14620 꽃길](https://www.acmicpc.net/problem/14620)

```c++
#include <iostream>
#include <stdio.h>
using namespace std;

int map[11][11];
int visit[11][11];
int price = 2147483647, len;



void dfs(int cnt, int pr){
    int dx[4] = {0, -1, 0, 1};
    int dy[4] = {-1, 0, 1, 0};
    int i, j, k, ax, ay, p;
    bool flag;
    
    if(cnt == 0){
        price = price < pr ? price : pr;
        return;
    }
    
    for(i = 1; i < len; i++){
        for(j = 1; j < len; j++){
            flag = true;
            if(visit[i][j] == 1)
                continue;
            for(k = 0; k < 4; k++){
                ax = i + dx[k];
                ay = j + dy[k];
                if(ax < 0 || ax >= len || ay < 0 || ay >= len || visit[ax][ay] == 1){
                    flag = false;
                    break;
                }
            }
            
            if(!flag)
                continue;
            
            visit[i][j] = 1;
            p = map[i][j];
            for(k = 0; k < 4; k++){
                ax = i + dx[k];
                ay = j + dy[k];
                visit[ax][ay] = 1;
                p += map[ax][ay];
            }
            
            dfs(cnt-1, pr+p);
            
            visit[i][j] = 0;
            for(k = 0; k < 4; k++){
                ax = i + dx[k];
                ay = j + dy[k];
                visit[ax][ay] = 0;
            }
        }
    }
}

int main(int argc, const char * argv[]) {

    int i = 0, j = 0;
    scanf("%d", &len);
    
    for(i = 0; i < len; i++){
        for(j = 0; j < len; j++){
            scanf("%d", &map[i][j]);
        }
    }

    dfs(3, 0);
    
    printf("%d", price);
    return 0;
}
```
